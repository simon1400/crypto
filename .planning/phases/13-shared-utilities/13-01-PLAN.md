export function validateTakeProfits(

6. Return `{ valid: true, error: null, parsed }`

export function calcPnlForecast(

export function calcTpPnl(

  type: 'LONG' | 'SHORT',
  leverage: number,
  amount: number,
  tpPercent: number
): { pct: number; usd: number }

**Function 2: `downloadCsv`** (extracted from the blob+download pattern shared across all 3 pages)
```typescript
export function downloadCsv(options: {
  header: string[]
  rows: (string | number)[][]
  filename: string
  separator?: string
}): void
```
Logic:
1. Default separator to ','
2. const esc = escapeCsvField
3. const headerLine = options.header.map(h => esc(h)).join(options.separator)
4. const dataLines = options.rows.map(r => r.map(v => esc(v)).join(options.separator))
5. const csv = '\uFEFF' + [headerLine, ...dataLines].join('\n')
6. const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' })
7. const url = URL.createObjectURL(blob)
8. const a = document.createElement('a')
9. a.href = url
10. a.download = options.filename
11. a.click()
12. URL.revokeObjectURL(url)

Note: The Signals.tsx CSV uses comma separator, while Scanner.tsx and Trades.tsx use semicolon (';'). The separator option handles both cases.
